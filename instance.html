# 인스턴스

인스턴스란...

자바스크립트에서의 인스턴스 제어

var div=document.querySelector('#app');

var obj={};
Object.defineProperty(obj, 'str', {
    get:function(){
        console.log('접근');
    }
    ,set:function(a){
        console.log('할당', a);
        div.innerHTML=a;
    }
});

Vue 를 이용한 인스턴스 제어

var vm=new Vue({
    el:'#app'
    ,data:{
        message:'HELLO VUE!!!'
    }
    ,methods:{
        showAlert:function(){
            alert('hi');
        }
    }
});
