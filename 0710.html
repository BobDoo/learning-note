# 3주차 수업 내용

  • ES6 문법

    • 변수 종류
    var a; // 기존 변수

let b; // let 은 한번 선언 시, 다시 선언 불가

const c; // const는 한번 할당한 값을 변경 불가

    • 변수의 유효범위

      • 기존 변수는 블록과 상관없이 전역에 영향을 미침
      • ES6 의 변수는 블록 단위로 영향을 미치는 범위를 제한 가능

    • 화살표 함수

      • 함수 정의 방식의 변화
      • 소스의 길이를 줄일 수 있다
      // 기존 함수
var a = function() {
  //...
};

// ES6 화살표 함수
var ()=>{
    //...
}

  • slot (슬롯)

    • 컴포넌트의 재사용성을 위한 기능
    • 소스의 구멍을 뚫어 다른 컴포넌트의 내용을 가져온다고 이해

  • Hackers News 게시판 실습

  • WebPack 환경 구성
